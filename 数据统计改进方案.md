```sql
ALTER TABLE dice_rolls ADD COLUMN session_time INTEGER; -- 会话时长（秒）
ALTER TABLE dice_rolls ADD COLUMN time_of_day TEXT;    -- 时段（早上/下午/晚上/深夜）

```

#### 1.2 新增统计数据收集点

**在 app.js 中增强统计：**
```javascript
// 增强的摇骰子统计
recordRoll(diceConfig, result, extra = {}) {
    const stats = {
        // 现有字段
        visitor_id: visitorId,
        dice_name: diceConfig.name,
        dice_id: diceConfig.id,
        result_word: result.word,
        result_emoji: result.emoji,
        result_desc: result.desc,

        // 新增字段
        mode_type: extra.mode || 'single',
        combo_id: extra.comboId || null,
        combo_name: extra.comboName || null,
        sub_dices: extra.subDices || null,

        // 时间维度
        session_time: this.getSessionTime(),
        time_of_day: this.getTimeOfDay(),
        day_of_week: this.getDayOfWeek(),

        // 用户行为
        roll_number_in_session: this.getRollNumberInSession(),
        time_on_page_before_roll: this.getTimeOnPage(),

        // 设备信息
        screen_width: window.screen.width,
        screen_height: window.screen.height,
        user_agent: navigator.userAgent,

        // 时间戳
        created_at: new Date().toISOString()
    };

    // 发送到Supabase
    SupabaseClient.insert('dice_rolls', stats);
}
```

### 方案三：数据导出功能

#### 3.1 添加导出按钮

```javascript
// 导出为CSV
exportToCSV() {
    const data = this.getAllRolls(); // 从Supabase获取
    const csv = this.convertToCSV(data);
    this.downloadCSV(csv, 'emotion-dice-data.csv');
}

// 导出为Excel（使用库）
exportToExcel() {
    // 使用 SheetJS (xlsx) 库
    const workbook = XLSX.utils.book_new();
    const worksheet = XLSX.utils.json_to_sheet(data);
    XLSX.utils.book_append_sheet(workbook, worksheet, "骰子数据");
    XLSX.writeFile(workbook, 'emotion-dice-data.xlsx');
}
```
